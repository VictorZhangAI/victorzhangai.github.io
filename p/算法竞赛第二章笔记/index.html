void findsum(int *a, int n, int s)
{
	int i = 0, j = 0;
	int sum = a[0];

	{
		if(sum >= s)

	}
}

int bin_search(int *a, int n, int x)
{
    int left = 0, right = n;
    while(left < right)
    {
        int mid = (left + right) / 2;
        if(a[mid] >= x) right = mid;
        else left = mid + 1;
    }
    return left;
}
